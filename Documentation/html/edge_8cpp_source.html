120 void Edge::adjust()
121 {
122  if (!source || !dest)
123  return;
124 
125  QLineF line(mapFromItem(source, source->Width() / 2, source->Height() / 2), mapFromItem(dest, dest->Width() / 2, dest->Height() / 2));
126  qreal length = line.length();
127 
128  prepareGeometryChange();
129 
130  qreal Ox, Oy, Dx, Dy;
131 
132  if (abs(line.dx()) < 1)
133  {
134  Ox = 0;
135  Dx = 0;
136  }
137  else
138  {
139  Ox = line.dx() / abs(line.dx())*min(int(source->Width() / 2), int(fabs(source->Height() / 2.0 * line.dx() / (line.dy()+0.5))));
140  Dx = -line.dx() / abs(line.dx())*min(int(dest->Width() / 2), int(fabs(dest->Height() / 2.0 * line.dx() / (line.dy()+0.5))));
141  }
142  if (abs(line.dy()) < 1)
143  {
144  Oy = 0;
145  Dy = 0;
146  }
147  else
148  {
149  Oy = line.dy() / abs(line.dy())*min(int(source->Height() / 2), int(fabs(source->Width()/ 2.0 * line.dy() / (line.dx()+0.5))));
150  Dy = -line.dy() / abs(line.dy())*min(int(dest->Height() / 2), int(fabs(dest->Width() / 2.0 * line.dy() / (line.dx()+0.5))));
151  }
152  QPointF edgeOffsetSource(Ox, Oy);
153  QPointF edgeOffsetDest(Dx, Dy);
154  sourcePoint = line.p1() + edgeOffsetSource;
155  destPoint = line.p2() + edgeOffsetDest;
156 }

250 int Edge::dist(const QPointF point)
251 {
252  int x1 = sourcePoint.x();
253  int y1 = sourcePoint.y();
254  int x2 = destPoint.x();
255  int y2 = destPoint.y();
256  int x0 = point.x();
257  int y0 = point.y();
258  int dist = abs(x0*(y2 - y1) - y0*(x2 - x1) + x2*y1 - y2*x1) / sqrt((y2 - y1) ^ 2 + (x2 - x1) ^ 2);
259  return dist;
